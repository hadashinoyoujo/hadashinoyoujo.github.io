# 448. Find All Numbers Disappeared in an Array

```c++
class Solution {
public:
    vector<int> findDisappearedNumbers(vector<int>& nums) {
        int n = nums.size();
        vector<int> res;
        map<int, int> sortedMap;
        for(int num : nums) {
            if(sortedMap.find(num) == sortedMap.end()) sortedMap[num] = 1;
            else ++sortedMap[num];
        }
        for(int i = 1; i <= n; ++i) {
            if(sortedMap.find(i) == sortedMap.end()) res.push_back(i);
        }
        return res;
    }
};
```
