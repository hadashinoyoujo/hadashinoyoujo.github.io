# 852. Peak Index in a Mountain Array

```c++
class Solution {
public:
    int peakIndexInMountainArray(vector<int>& arr) {
        int res = 0, left = 0, right = arr.size() - 1;
        while(left < arr.size() - 1 && arr[left] < arr[left + 1])
            ++left;
        while(right > 0 && arr[right - 1] > arr[right])
            --right;
        if(left == right)
            return left;
        return 0;
    }
};
```
