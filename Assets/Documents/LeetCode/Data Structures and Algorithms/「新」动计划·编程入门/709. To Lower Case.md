# 709. To Lower Case

```c++
class Solution {
public:
    string toLowerCase(string s) {
        int n = s.size();
        for(int i = 0; i < n; ++i) {
            if(s[i] - 'A' <= 25 && s[i] - 'A' >= 0) 
                s[i] = s[i] + 32;
            else
                continue;
        }
        return s;
    }
};
```
