# 1422. Maximum Score After Splitting a String

```c++
class Solution {
public:
    int maxScore(string s) {
        int n = s.size(), max = 0, zeroLeft = 0, oneRight = 0;
        for(int i = 0; i < n - 1; ++i) {
            if(s[i] == '0') ++zeroLeft;
            oneRight = count(s.begin() + i + 1, s.end(), '1');
            max = (zeroLeft + oneRight) > max ? (zeroLeft + oneRight) : max;
        }

        return max;
    }
};
```
