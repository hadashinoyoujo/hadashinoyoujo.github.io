# 1299. Replace Elements with Greatest Element on Right Side

## Method One

```c++
class Solution {
public:
    vector<int> replaceElements(vector<int>& arr) {
        int n = arr.size(), max = -1, curr = INT_MIN;
        for(int i = n - 1; i >= 0; --i) {
            max = max > curr ? max : curr;
            curr = arr[i];
            arr[i] = max;
        }
        return arr;
    }
};
```
