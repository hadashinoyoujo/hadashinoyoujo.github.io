# 14. Longest Common Prefix

Beats 100%

```c++
class Solution {
public:
    string longestCommonPrefix(vector<string>& strs) {
        string res = "";
        int minString = INT_MAX, n = strs.size();
        char c;

        for(string s : strs) {
            minString = std::min(minString, (int)s.size());
        }

        for(int i = 0; i < minString; ++i) {
            c = strs[0][i];
            for(int j = 1; j < n; ++j) {
                if(strs[j][i] != c) return res;
            }
            res.push_back(c);
        }
        return res;
    }
};
```
