# 26. Remove Duplicates from Sorted Array

```c++
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int n = nums.size();
        if(n == 1) return 1;
        int slow = 0, fast = 0, lastNum = INT_MAX;
        
        while(fast < n) {
            if(nums[fast] != lastNum) {
                nums[slow] = nums[fast];
                lastNum = nums[slow];
                ++slow;
            }
            ++fast;
        }
        return slow;
    }
};
```
