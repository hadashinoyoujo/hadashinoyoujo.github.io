# 643. Maximum Average Subarray I

```c++
class Solution {
public:
    double findMaxAverage(vector<int>& nums, int k) {
        int n = nums.size(), max = INT_MIN, count = 0;
        for(int i = 0, j = k - 1; j < n; ++i, ++j) {
            if(i == 0) {
                for(; i < k; ++i) {
                    count += nums[i];
                };
                i = 0;
                max = count;
            }
            else {
                count = count - nums[i - 1] + nums[j];
                max = std::max(max, count);
            }
        }
        return (double)max / k;
    }
};
```

注意限定条件。
