# 2461. Maximum Sum of Distinct Subarrays With Length K

```c++
class Solution {
public:
    long long maximumSubarraySum(vector<int>& nums, int k) {
        int n = nums.size(), i = 0;
        long long sum = 0, max = 0;
        unordered_map<int, int> map;
        while(i < k && i < n) {
            ++map[nums[i]];
            sum += nums[i];
            ++i;
        }
        if(map.size() == k) max = sum;
        
        while(i < n) {
            ++map[nums[i]];
            --map[nums[i - k]];
            if(map[nums[i - k]] == 0) map.erase(nums[i - k]);

            sum += nums[i];
            sum -= nums[i - k];
            ++i;
            if(map.size() == k) max = std::max(max, sum);
        }
        return max;
    }
};
```
