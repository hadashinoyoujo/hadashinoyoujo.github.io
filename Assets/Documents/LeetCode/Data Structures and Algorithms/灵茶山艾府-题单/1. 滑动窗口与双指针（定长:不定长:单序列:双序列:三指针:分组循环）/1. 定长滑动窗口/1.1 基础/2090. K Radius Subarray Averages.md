# 2090. K Radius Subarray Averages

```c++
class Solution {
public:
    vector<int> getAverages(vector<int>& nums, int k) {
        if(k == 0) return nums;

        int n = nums.size();
        long long sum = 0;
        bool flag = false;
        vector<int> res;
        for(int i = 0; i < n; ++i) {
            if(i - k < 0 || i + k >= n) {
                res.push_back(-1);
                continue;
            }
            else {
                if(flag == false) {
                    for(int left = i - k, right = i + k; left <= right; ++left) {
                        sum += nums[left];
                    }
                    res.push_back(sum / (2 * k + 1));
                    flag = true;
                }
                else {
                    sum -= nums[i - k - 1];
                    sum += nums[i + k];
                    res.push_back(sum / (2 * k + 1));
                }
            }
        }
        return res;
    }
};
```
