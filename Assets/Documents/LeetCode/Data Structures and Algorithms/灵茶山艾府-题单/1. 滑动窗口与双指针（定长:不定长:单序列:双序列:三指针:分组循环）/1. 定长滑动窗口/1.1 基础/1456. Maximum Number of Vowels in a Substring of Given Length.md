# 1456. Maximum Number of Vowels in a Substring of Given Length

```c++
class Solution {
public:
    bool isVowel(char c) {
        if(c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u')
            return true;
        else 
            return false;
    }
    int maxVowels(string s, int k) {
        int n = s.size(), countVowels = 0, maxVowels = 0;
        for(int left = 0, right = k - 1; right < n; ++left, ++right) {
            if(left == 0) {
                for(; left < k; ++left) {
                    if(isVowel(s[left])) ++countVowels;
                }
                left = 0;
                maxVowels = maxVowels > countVowels ? maxVowels : countVowels;
            }
            else {
                if(isVowel(s[left - 1])) --countVowels;
                if(isVowel(s[right])) ++countVowels;
                maxVowels = maxVowels > countVowels ? maxVowels : countVowels;
            }
        }
        return maxVowels;
    }
};
```
