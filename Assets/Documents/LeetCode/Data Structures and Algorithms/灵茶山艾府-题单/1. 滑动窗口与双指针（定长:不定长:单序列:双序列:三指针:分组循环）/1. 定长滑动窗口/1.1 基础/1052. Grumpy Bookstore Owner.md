# 1052. Grumpy Bookstore Owner

Beats 100%。看着唬人，实际就是定长滑动窗口的小改款，没有本质区别。

```c++
class Solution {
public:
    int maxSatisfied(vector<int>& customers, vector<int>& grumpy, int minutes) {
        int n = customers.size();
        int basic = 0, additional = 0, maxAdditional = INT_MIN;
        vector<int> anotherVec;
        for(int i = 0; i < n; ++i) {
            if(grumpy[i] == 0) {
                basic += customers[i];
                anotherVec.push_back(0);
            }
            else {
                anotherVec.push_back(customers[i]);
            }
        }

        // 统计滑动窗口能获取到的最多的顾客。
        for(int i = 0, j = minutes - 1; j < n; ++i, ++j) {
            if(i == 0) {
                for(; i < minutes; ++i) {
                    additional += anotherVec[i];
                }
                i = 0;
            }
            else {
                additional -= anotherVec[i - 1];
                additional += anotherVec[j];
            }

            maxAdditional = std::max(maxAdditional, additional);
        }

        return basic + maxAdditional;
    }
};
```
