# 2379. Minimum Recolors to Get K Consecutive Black Blocks

```c++
class Solution {
public:
    int minimumRecolors(string blocks, int k) {
        int n = blocks.size(), count = 0, min = INT_MAX;
        for(int i = 0, j = k - 1; j < n; ++i, ++j) {
            if(i == 0) {
                for(; i <= k - 1; ++i) {
                    if(blocks[i] == 'W') ++count;
                }
                if(count == 0) return 0;
                i = 0;
                min = std::min(min, count);
            }
            else {
                if(blocks[i - 1] == 'W') --count;
                if(blocks[j] == 'W') ++count;
                min = std::min(min, count);
            }
        }
        return min;
    }
};
```
