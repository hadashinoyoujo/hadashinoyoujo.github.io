# 1423. Maximum Points You Can Obtain from Cards

```c++
class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        int n = cardPoints.size(), sum = 0, max = 0, count = 0;
        int i = k - 1, j = n;
        // 将 cardPoints 看作环形数组，求滑动窗口的最大值。
        // 从牌组顶部或者底部抽卡，意味着该窗口至少包含 下标为 0 或者 n - 1 二者之一的元素。
        for(; i >= 0; --i, --j) {
            if(i == k - 1) {
                while(i >= 0) {
                    sum += cardPoints[i];
                    --i;
                }
                i = k - 1;
                max = std::max(max, sum);
            }
            else if(i >= 0) {
                sum -= cardPoints[i + 1];
                sum += cardPoints[j];
                max = std::max(max, sum);
            }
        }
        sum -= cardPoints[0];
        sum += cardPoints[j];
        max = std::max(max, sum);

        return max;
    }
};
```
