# 20. Valid Parentheses

Beats 100%

```c++
class Solution {
public:
    bool isValid(string s) {
        int n = s.size();
        if(n % 2 != 0) return false;
        stack<char> stackChar;
        for(int i = 0; i < n; ++i) {
            switch(s[i]) {
                case '(':
                    stackChar.push(')');
                    break;
                case ')':
                    if(stackChar.empty()) return false;
                    else if(stackChar.top() != ')') return false;
                    stackChar.pop();
                    break;
                case '{':
                    stackChar.push('}');
                    break;
                case '}':
                    if(stackChar.empty()) return false;
                    else if(stackChar.top() != '}') return false;
                    stackChar.pop();
                    break;
                case '[':
                    stackChar.push(']');
                    break;
                case ']':
                    if(stackChar.empty()) return false;
                    else if(stackChar.top() != ']') return false;
                    stackChar.pop();
                    break;
                default:
                    std::cout << "Error" << std::endl;
                    break;
            }
        }
        if(stackChar.empty()) return true;
        else return false;
    }
};
```

注意遇到右括号先判断栈是否为空，不为空才能使用 top 函数。
