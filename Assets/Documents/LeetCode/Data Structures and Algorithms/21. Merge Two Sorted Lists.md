# 21. Merge Two Sorted Lists

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        ListNode dummy;
        ListNode* res = &dummy;
        ListNode* curr = res;
        while(list1 != nullptr && list2 != nullptr) {
            if(list1->val <= list2->val) {
                curr->next = list1;
                curr = curr->next;
                list1 = list1->next;
            }
            else {
                curr->next = list2;
                curr = curr->next;
                list2 = list2->next;
            }
        }
        curr->next = list1 == nullptr ? list2 : list1;
        return res->next;
    }
};
```

### Attention

1. Return the abstract node's next pointer.
2. Dummy node can not be omitted, because res should point to it, and ensure the area which pointed is a node, not a unknown area.
