# 414. Third Maximum Number

```c++
class Solution {
public:
    int thirdMax(vector<int>& nums) {
        int n = nums.size(), thirdMax = INT_MAX, count = 3;
        std::sort(nums.begin(), nums.end(), std::greater<int>());
        if(nums[0] == INT_MAX) --count;
        for(int i = 0; i < n; ++i) {
            if(thirdMax > nums[i]) {
                thirdMax = nums[i];
                --count;
            }
            if(count == 0) return thirdMax;
        }
        return nums[0];
    }
};
```
