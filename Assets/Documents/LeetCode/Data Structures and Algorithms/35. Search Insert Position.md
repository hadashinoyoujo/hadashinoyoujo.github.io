# 35. Search Insert Position

Beats 100%

```c++
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int n = nums.size(), curr = 0, prev = 0;
        for(; curr < n; ++curr) {
            if(nums[curr] >= target) return curr;
        }
        return n;
    }
};
```
